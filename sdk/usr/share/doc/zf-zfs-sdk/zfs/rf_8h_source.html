257  */
258  inline uint16_t rf24to16(uint32_t rf)
259  {
260  if (rf > 65535)
261  return static_cast<uint16_t>((rf >> 8) | 1);
262  else
263  return static_cast<uint16_t>(rf & 0xfffe);
264  }

266  /**
267  @brief convert the compressed intensity as it is stored (16bit) to a real number (24bit)
268  */
269  inline uint32_t rf16to24(uint16_t rf)
270  {
271  if (rf & 1)
272  return static_cast<uint32_t>(rf & ~1) << 8;
273  else
274  return static_cast<uint32_t>(rf);
